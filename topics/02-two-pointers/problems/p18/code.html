class Solution {
public:
    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        int n = nums.size();
        sort(nums.begin(), nums.end());
        vector<vector<int>> ans;

        for (int i = 0; i < n; i++) {
            if (i > 0 && nums[i] == nums[i - 1]) continue;       // 去重 i

            for (int j = i + 1; j < n; j++) {
                if (j > i + 1 && nums[j] == nums[j - 1]) continue;  // 去重 j

                int l = j + 1, r = n - 1;

                while (l < r) {
                    long long sum = (long long)nums[i] + nums[j] + nums[l] + nums[r];  // 防溢位

                    if (sum < target) {
                        l++;
                    } else if (sum > target) {
                        r--;
                    } else {
                        ans.push_back({nums[i], nums[j], nums[l], nums[r]});
                        l++; r--;
                        while (l < r && nums[l] == nums[l - 1]) l++;  // 去重 l
                        while (l < r && nums[r] == nums[r + 1]) r--;  // 去重 r
                    }
                }
            }
        }

        return ans;
    }
};
和 3Sum 相比,骨架只多了「次外層 j 迴圈 + 一處去重」。內層那三分支(l++/r--/收下)一字不改。 唯一容易踩的雷是 long long:(long long)nums[i] + nums[j] + ... —— 只要第一個轉型,整條運算就在 long long 下進行,不會中途以 int 溢位。

溢位是怎麼發生的// THE OVERFLOW

// nums[i] ∈ [-1e9, 1e9],int 上限 ≈ 2.147e9 最大四數和 ≈ 4 × 10⁹ > INT_MAX(2 147 483 647) → int 溢位、變負數 → 比較亂 // 錯:中途以 int 相加 int sum = nums[i] + nums[j] + nums[l] + nums[r]; // nums[i]+nums[j] 已可能溢位 // 對:第一個轉 long long,後面自動提升 long long sum = (long long)nums[i] + nums[j] + nums[l] + nums[r]; 關鍵:轉型要放在「第一次加法之前」,否則 int+int 已經溢位才轉就來不及

C++ 的加法由左往右結合:((nums[i]+nums[j])+nums[l])+nums[r]。若沒轉型,nums[i]+nums[j] 這第一步就是 int+int,可能已溢位。 把最左邊那個 operand 轉 long long,之後每一步都變 long long + int → long long,全程安全。這是所有「多數相加」題的通用護身符。

邊界與陷阱// PITFALLS

陷阱症狀正解
int 累加溢位 四個 ~1e9 相加超過 INT_MAX → 比較錯 (long long)nums[i] + …
j 去重條件寫錯 用 j>0 → 每個 i 的第一個 j 被誤跳 j > i+1 && nums[j]==nums[j-1]
忘了 j 那層去重 相同 nums[j] → 重複四元組 四處去重缺一不可
命中後跳重複用錯索引 比 nums[l+1] 而非 nums[l-1] 已先 l++,比剛用過的 nums[l-1]
target 是負數 擔心比較出錯 兩邊都在 long long 下比,沒問題

細節走讀// WALKTHROUGH

// nums=[1,0,-1,0,-2,2] → 排序 [-2,-1,0,0,1,2], target=0 i=0(-2), j=1(-1): 內層找 l+r = 0-(-2)-(-1)=3 l=2,r=5: -2-1+0+2=-1<0→l++ ... l=4,r=5: -2-1+1+2=0 → 收下 [-2,-1,1,2] i=0(-2), j=2(0): 內層找 l+r = 2 l=3,r=5: -2+0+0+2=0 → 收下 [-2,0,0,2] i=0(-2), j=3(0): nums[3]==nums[2] → continue(去重 j) i=1(-1), j=2(0): l=3,r=5: -1+0+0+1... → 收下 [-1,0,0,1] i=2(0) 起:最小四數和已 ≥ 0 且無法再湊 → 收尾 ans = [[-2,-1,1,2], [-2,0,0,2], [-1,0,0,1]]
i=0, j=3 那步正是 j 去重的作用:第二個 0 若不跳,會重複產生 [-2,0,0,2]。 四層去重(i、j、l、r)各守一個位置,合起來保證答案無重複 —— 這是 4Sum 相對 3Sum 唯一「多記」的地方。
