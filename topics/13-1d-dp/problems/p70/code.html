寫法 A · dp 陣列// BOTTOM-UP

p70.cpp C++17 
class Solution {
public:
    int climbStairs(int n) {
        vector<int> dp(n + 1);
        dp[0] = dp[1] = 1;              // base:0 階、1 階各 1 種

        for (int i = 2; i <= n; i++)
            dp[i] = dp[i - 1] + dp[i - 2]; // 最後一步跨 1 或跨 2

        return dp[n];
    }
};

寫法 B · 滾動兩變數// O(1) SPACE

p70_o1.cpp C++17 
class Solution {
public:
    int climbStairs(int n) {
        int prev2 = 1, prev1 = 1;          // dp[i-2]、dp[i-1]
        for (int i = 2; i <= n; i++) {
            int cur = prev1 + prev2;          // dp[i]
            prev2 = prev1;                      // 往前滾一格
            prev1 = cur;
        }
        return prev1;
    }
};

因為 dp[i] 只看 前兩項,整條陣列其實用不到 —— 兩個變數 prev1/prev2 一路往前滾就夠,空間降到 O(1)。這是一維 DP 最常見的「滾動陣列」優化。
