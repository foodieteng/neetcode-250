寫法 A · dp 陣列// BOTTOM-UP

class Solution {
public:
    int tribonacci(int n) {
        if (n == 0) return 0;              // 特判:避免 n≤2 時存取不到的位置
        if (n <= 2) return 1;

        vector<int> t(n + 1, 0);
        t[0] = 0; t[1] = t[2] = 1;        // 三個 base

        for (int i = 3; i <= n; i++)
            t[i] = t[i - 3] + t[i - 2] + t[i - 1]; // 前三項之和

        return t[n];
    }
};

寫法 B · 滾動三變數// O(1) SPACE

p1137_o1.cpp C++17 
class Solution {
public:
    int tribonacci(int n) {
        if (n == 0) return 0;
        if (n <= 2) return 1;
        int a = 0, b = 1, c = 1;            // t[i-3]、t[i-2]、t[i-1]
        for (int i = 3; i <= n; i++) {
            int cur = a + b + c;            // t[i]
            a = b; b = c; c = cur;          // 三個一起往前滾
        }
        return c;
    }
};

和 70 的滾動優化同理,只是留三個變數(費氏留兩個)。a→b→c→cur 依序往前推,順序不能亂,不然會用到被覆蓋過的舊值。
