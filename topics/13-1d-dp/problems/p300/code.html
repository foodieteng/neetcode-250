寫法 A · tails + lower_bound// O(n log n)

p300.cpp C++17 
class Solution {
public:
    int lengthOfLIS(vector<int>& nums) {
        vector<int> tails;              // tails[k] = 長度 k+1 的最小結尾

        for (int num : nums) {
            auto it = lower_bound(tails.begin(), tails.end(), num);
            if (it == tails.end()) tails.push_back(num);  // num 最大 → 接長
            else *it = num;                          // 覆蓋:更小的結尾
        }

        return tails.size();
    }
};

tails 永遠遞增,所以能二分。lower_bound 找第一個 ≥ num 的位置:找不到(num 比全部大)就 接長;找到就覆蓋(把某長度的結尾換成更小的 num,更容易被後面接續)。tails.size() 即 LIS 長度 —— 但 tails 的內容不保證是一條真正的 LIS。

寫法 B · O(n²) DP(直覺版)// END AT i

p300_dp.cpp C++17 
int lengthOfLIS(vector<int>& nums) {
    int n = nums.size();
    vector<int> dp(n, 1);              // dp[i]:以 i 結尾的 LIS,至少 1

    for (int i = 0; i < n; i++) {
        for (int j = i - 1; j >= 0; j--) {
            if (nums[i] > nums[j])       // 可接在 j 後面
                dp[i] = max(dp[i], dp[j] + 1);
        }
    }

    return *max_element(dp.begin(), dp.end());   // LIS 可結尾於任何 i
}

直覺、好推導,面試常先寫這個。注意回傳是 max(dp) 不是 dp[n−1] —— LIS 可能在中間某個 i 就結束。n ≤ 2500 時 O(n²) ≈ 6×10⁶ 綽綽有餘。

p300_nondecr.cpp C++17 
// 最長「非遞減」子序列(允許相等)：把 lower_bound 換成 upper_bound
int longestNonDecreasing(vector<int>& nums) {
    vector<int> tails;
    for (int num : nums) {
        auto it = upper_bound(tails.begin(), tails.end(), num); // 第一個 > num
        if (it == tails.end()) tails.push_back(num);
        else *it = num;
    }
    return tails.size();
}
