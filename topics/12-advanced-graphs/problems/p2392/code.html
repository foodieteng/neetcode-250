class Solution {
public:
    vector<vector<int>> buildMatrix(int k, vector<vector<int>>& rowConditions,
                                       vector<vector<int>>& colConditions) {
        vector<int> rowOrder = topo(rowConditions, k);   // 列順序
        vector<int> colOrder = topo(colConditions, k);   // 行順序
        if (rowOrder.empty() || colOrder.empty()) return {};  // 有環 → 無解

        vector<pair<int,int>> pos(k + 1, {-1, -1});
        for (int i = 0; i < k; i++) {
            pos[rowOrder[i]].first  = i;          // 該數字的列索引
            pos[colOrder[i]].second = i;          // 該數字的行索引
        }

        vector<vector<int>> matrix(k, vector<int>(k, 0));
        for (int v = 1; v <= k; v++)
            matrix[pos[v].first][pos[v].second] = v;   // 交叉放格
        return matrix;
    }

    // Kahn 拓樸排序:回傳長度 k 的順序,若有環回傳空
    vector<int> topo(vector<vector<int>>& arr, int k) {
        vector<vector<int>> graph(k + 1);
        vector<int> indegree(k + 1, 0);
        for (auto e : arr) {
            auto u = e[0], v = e[1];
            graph[u].push_back(v);                     // u 在 v 之前(上/左)
            indegree[v]++;
        }

        vector<int> order;
        queue<int> q;
        for (int i = 1; i <= k; i++)
            if (indegree[i] == 0) q.push(i);       // 入度 0 先入隊

        while (!q.empty()) {
            auto u = q.front(); q.pop();
            order.push_back(u);
            for (auto v : graph[u])
                if (--indegree[v] == 0) q.push(v);   // 入度歸零才入隊
        }
        return order.size() == k ? order : vector<int>();  // 少於 k = 有環
    }
};

步驟成本說明
建圖 + 入度O(k + E)E = 條件筆數
Kahn ×2O(k + E)每點每邊各處理一次
建 posO(k)掃兩個順序陣列
初始化 + 填矩陣O(k²)k×k 全 0 再放 k 個值
T = O(k² + E)　　S = O(k² + E)

k = 數字個數(≤ 400),E = rowConditions + colConditions 筆數(≤ 2×10⁴)。矩陣本身佔 O(k²)。

邊界與陷阱// PITFALLS

陷阱後果正解
把 row / col 條件併成一張圖 製造假環 行、列是兩個獨立維度,必須各自拓樸
忘了判環 回傳壞矩陣 拓樸 order.size() != k 即有環 → 回傳 {}
陣列開到 k 而非 k+1 值 k 越界 數字是 1..k,以值當索引要開 k + 1
誤以為要「唯一解」 過度約束 題目要「任一」合法解,拓樸的任一有效順序都行
擔心兩數字撞到同格 不會發生 rowPos、colPos 皆為排列,k 個值必落相異格

// EDGE: [a, b] 為什麼是 a → b rowConditions [a, b]:a 要在 b「上面」= a 的列比 b 小 拓樸排序保證 a 排在 b 之前 ⇒ 先出隊的拿到較小索引 所以建邊 graph[a].push_back(b)、indegree[b]++ ↳ 入度 = 「還有幾個數字必須排在我前面」
// KAHN: 為什麼消入度能排出拓樸序 入度 0 = 沒有前置限制 → 可立刻放到最前面 每放一個 u,它指向的 v 少一個前置 → --indegree[v] indegree[v] 歸零表示 v 的前置全部就位 → 入隊 若還有點永遠進不了隊(入度卡著)⇒ 存在環
// CYCLE: order.size() != k 的意義 無環時每個點都會恰好入隊、出隊一次 ⇒ order 長度 = k 有環時環上的點入度永遠 > 0,進不了隊 ⇒ order < k 例:1→2→3→1,三點入度都是 1,初始沒人能入隊 → order 空
