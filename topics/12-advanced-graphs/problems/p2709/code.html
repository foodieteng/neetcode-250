class Solution {
private:
    vector<int> parent;        // 負值 = 該根的分量大小（union by size）

    int Find(int x) {
        return parent[x] < 0 ? x : parent[x] = Find(parent[x]);  // 路徑壓縮
    }

    bool Union(int x, int y) {
        int rootX = Find(x), rootY = Find(y);
        if (rootX == rootY) return false;
        if (-parent[rootX] < -parent[rootY]) swap(rootX, rootY);  // rootX 是較大分量
        parent[rootX] += parent[rootY];
        parent[rootY] = rootX;
        return true;
    }

public:
    bool canTraverseAllPairs(vector<int>& nums) {
        int n = nums.size();
        if (n == 1) return true;              // 單一元素永遠自連通

        int mx = *max_element(nums.begin(), nums.end());
        if (mx == 1) return false;            // n>1 但全是 1 ⇒ 兩兩 gcd=1

        parent.assign(mx + 1, -1);
        vector<int> spf(mx + 1);
        iota(spf.begin(), spf.end(), 0);       // spf[i] 初值 = i

        // 篩最小質因數：只在 spf[j] 還沒被更小質數蓋過時才寫入
        for (int i = 2; i * i <= mx; i++) {
            if (spf[i] == i) {                 // i 是質數
                for (int j = i * i; j <= mx; j += i)
                    if (spf[j] == j) spf[j] = i;
            }
        }

        int anchor = -1;
        for (int v : nums) {
            if (v == 1) return false;          // 含 1 且 n>1 ⇒ 走不到
            int prev = -1, p = spf[v];
            while (v != 1) {                // 逐一剝掉質因數
                if (prev != -1) Union(prev, p);  // 把 v 的質因數併成一組
                else if (anchor == -1) anchor = p;  // 記第一個看到的質數
                prev = p;
                v /= p;
                p = spf[v];
            }
        }

        for (int v : nums)                    // 連通性檢查：用質因數 spf[v]，非合成數 v
            if (Find(anchor) != Find(spf[v])) return false;
        return true;
    }
};

複雜度分析// COMPLEXITY

步驟成本說明
iota + assignO(A)兩個長度 max+1 的陣列
篩 spf[]O(A log log A)埃氏篩，只改未被覆蓋者
拆質因數 + unionO(n log A · α)每數 ≤ log₂A 個質因數
連通性檢查O(n · α)每數一次 Find
T = O(A log log A + n log A)　　S = O(A)

A = 陣列最大值（≤ 10⁵），n = 陣列長度。α 為反 Ackermann，視為常數。

維度量級
TimeO(A log log A + n log A)
SpaceO(A)

邊界與陷阱// PITFALLS

陷阱後果正解
篩的守門寫成 spf[j] == i 篩完全失效 應為 spf[j] == j：只在 j 仍是「自己」（尚未被更小質數蓋過）時寫入
檢查用合成數 Find(v) 越界 / 錯根 union 的是質數，檢查也要用質因數 Find(spf[v])
單一元素 n == 1 誤判 false 先行 return true（沒有任何「一對」需要連）
陣列含 1 且 n > 1 1 與任何數 gcd=1 迴圈遇到 v == 1 立即 return false
parent/spf 只開到 n 以質數值當索引會越界 兩者都開到 max(nums) + 1

// SIEVE: 為什麼守門是 spf[j] == j spf 初值：spf[i] = i（每個數先假設自己是最小質因數） 外層 i 由小到大；i 是質數 ⇔ spf[i] 還等於 i（沒被更小質數改過） 內層把 i 的倍數 j 標記 spf[j] = i，但只在 spf[j] == j 時寫 ↳ 這樣「先到先得」，最小的質數 i 先蓋，之後大質數不再覆蓋 ↳ 若誤寫成 spf[j] == i，條件幾乎永遠不成立 → 篩失效，spf 全留 j
// FACTORIZE: while (v != 1) 逐層剝質因數 p = spf[v] 取當前最小質因數 prev 記「上一個質因數」；prev != -1 時 Union(prev, p) ↳ 把 v 的相鄰質因數串起來 ⇒ v 的所有質因數同一組 anchor 只在「整個過程第一次遇到質數」時設定一次 v /= p 後重取 spf[v]，重複到 v 變 1 例：v=12 → p=2, v=6 → p=2(Union 2,2 無效), v=3 → p=3 Union(2,3), v=1 停
