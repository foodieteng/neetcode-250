寫法 A · Prim(lazy · min-heap)// GROW THE TREE

p1584_prim.cpp C++17 
class Solution {
public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        auto manhattan = [&](int u, int v) {          // 兩點成本 = 曼哈頓距離
            return abs(points[u][0] - points[v][0]) + abs(points[u][1] - points[v][1]);
        };

        int n = points.size();
        vector<bool> inMST(n, false);            // 已在生成樹中的點
        using P = pair<int, int>;              // (權重, 目標點)
        priority_queue<P, vector<P>, greater<P>> pq;  // min-heap

        pq.push({0, 0});                        // 從 0 號點、成本 0 起長樹
        int total = 0, cnt = 0;

        while (!pq.empty() && cnt < n) {
            auto [w, u] = pq.top(); pq.pop();
            if (inMST[u]) continue;              // lazy:過期條目直接略過

            inMST[u] = true;                   // u 併入樹
            total += w; cnt++;

            for (int v = 0; v < n; v++) {
                if (inMST[v] || u == v) continue;
                pq.push({manhattan(u, v), v});   // u→樹外各點的跨邊入堆
            }
        }
        return total;
    }
};

寫法 B · Kruskal(Union-Find)// PICK SMALLEST EDGE

p1584_kruskal.cpp C++17 
class Solution {
private:
    vector<int> parent;                       // parent[x]<0:根,值 = −集合大小

    int Find(int x) {                         // 路徑壓縮
        return parent[x] < 0 ? x : parent[x] = Find(parent[x]);
    }

    bool Union(int x, int y) {
        int rootX = Find(x), rootY = Find(y);
        if (rootX == rootY) return false;       // 已同集合 → 成環,不合併
        if (-parent[rootX] < -parent[rootY]) swap(rootX, rootY); // 按大小合併
        parent[rootX] += parent[rootY];
        parent[rootY] = rootX;
        return true;
    }

public:
    int minCostConnectPoints(vector<vector<int>>& points) {
        auto manhattan = [&](int u, int v) {
            return abs(points[u][0] - points[v][0]) + abs(points[u][1] - points[v][1]);
        };

        int n = points.size();
        parent.assign(n, -1);
        using T = tuple<int, int, int>;         // (權重, u, v)
        auto cmp = [&](auto t1, auto t2) {
            return get<0>(t1) > get<0>(t2);    // 依權重做 min-heap
        };
        priority_queue<T, vector<T>, decltype(cmp)> pq(cmp);

        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                pq.push({manhattan(i, j), i, j});   // 所有點對的邊
            }
        }

        int total = 0, add = 0;
        while (add < n - 1) {                    // 加滿 n−1 條邊即完成
            auto [w, u, v] = pq.top(); pq.pop();
            if (Union(u, v)) {                   // 不成環才收
                total += w; add++;
            }
        }
        return total;
    }
};

細節走讀// WALKTHROUGH

// A · 為什麼 Prim 每步挑「最小跨邊」是對的 切分定理(cut property):把點分成「樹內」與「樹外」兩群 橫跨兩群的所有邊裡,最小的那條一定屬於某棵 MST min-heap 每次彈出的正是這條最小跨邊 → 貪心安全 lazy:同一個點可能被多條邊推入堆,彈到已 inMST 的就丟
// B · 為什麼 Kruskal 由小到大加、成環就跳 邊照權重由小到大考慮 兩端不同集合 → 這條邊連起兩塊、不可能更省 → union 收下 兩端已同集合 → 收了會成環(多餘)→ 丟棄 加滿 n−1 條 → n 個點恰好連成一棵樹 → 完成
// Union-Find · parent 的負數技巧 parent[x] < 0 表示 x 是根,其值 = −(該集合大小) Find 用三元 + 路徑壓縮:parent[x] = Find(parent[x]) Union 按大小掛小樹到大樹(-parent 比較大小),樹更扁 回傳 bool:true = 真的合併了,false = 早已同根
PREV概念與動畫 NEXT複習與 Trace
// TWO WAYS, ONE MST
Prim 從點長樹、Kruskal 從邊挑起,都得同一棵最小生成樹,皆 O(n²·log n)。
