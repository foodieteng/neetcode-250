寫法 A · 上升水位 + Union-Find// RISING WATER

p778_uf.cpp C++17 
class Solution {
private:
    vector<int> parent;

    int Find(int x) {
        return parent[x] < 0 ? x : parent[x] = Find(parent[x]);
    }

    void Union(int x, int y) {
        int a = Find(x), b = Find(y);
        if (a == b) return;
        if (-parent[a] < -parent[b]) swap(a, b);
        parent[a] += parent[b];
        parent[b] = a;
    }

public:
    int swimInWater(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        auto getId = [&](int r, int c) { return r * n + c; };
        parent.assign(m * n, -1);

        const vector<pair<int, int>> dirs = {{-1,0},{1,0},{0,-1},{0,1}};
        vector<bool> on(m * n, false);

        using P = pair<int, int>;
        priority_queue<P, vector<P>, greater<P>> pq;   // (高度, id)
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                pq.push({grid[i][j], getId(i, j)});
            }
        }

        while (!pq.empty()) {
            auto [h, id] = pq.top();
            pq.pop();
            on[id] = true;                    // 淹沒這格
            int r = id / n, c = id % n;

            for (auto [dr, dc] : dirs) {
                int nr = r + dr, nc = c + dc;
                if (nr < 0 || nc < 0 || nr >= m || nc >= n) continue;
                int nid = getId(nr, nc);
                if (on[nid]) Union(id, nid);   // 與已淹鄰居連通
            }

            if (Find(0) == Find(getId(m - 1, n - 1))) {
                return h;                    // 起終點連通 → 答案
            }
        }
        return -1;
    }
};

class Solution {
public:
    int swimInWater(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        using T = tuple<int, int, int>;            // (瓶頸, r, c)
        priority_queue<T, vector<T>, greater<T>> pq;

        const vector<pair<int, int>> dirs = {{-1,0},{1,0},{0,-1},{0,1}};
        vector<vector<int>> dist(m, vector<int>(n, -1));

        dist[0][0] = grid[0][0];
        pq.emplace(grid[0][0], 0, 0);

        while (!pq.empty()) {
            auto [h, r, c] = pq.top();
            pq.pop();

            if (dist[r][c] != -1 && h > dist[r][c]) continue;   // 過期條目
            if (r == m - 1 && c == n - 1) return h;

            for (auto [dr, dc] : dirs) {
                int nr = r + dr, nc = c + dc;
                if (nr < 0 || nr >= m || nc < 0 || nc >= n) continue;

                int d = max(h, grid[nr][nc]);          // 成本 = 沿途最大高度
                if (dist[nr][nc] == -1 || dist[nr][nc] > d) {
                    dist[nr][nc] = d;
                    pq.emplace(d, nr, nc);
                }
            }
        }
        return -1;
    }
};

class Solution {
public:
    int swimInWater(vector<vector<int>>& grid) {
        int n = grid.size();
        int l = max(grid[0][0], grid[n - 1][n - 1]);   // 下界:起終點高度
        int r = n * n - 1;                              // 上界:最大高度

        while (l < r) {
            int mid = l + (r - l) / 2;
            if (bfs(grid, mid)) {
                r = mid;                                 // 可達 → 縮小上界
            } else {
                l = mid + 1;                             // 不可達 → 抬高下界
            }
        }
        return l;
    }

private:
    bool bfs(const vector<vector<int>>& grid, int t) {
        int n = grid.size();
        if (grid[0][0] > t) return false;

        queue<pair<int, int>> q;
        vector<vector<bool>> seen(n, vector<bool>(n, false));
        q.push({0, 0});
        seen[0][0] = true;

        const vector<pair<int, int>> dirs = {{-1,0},{1,0},{0,-1},{0,1}};

        while (!q.empty()) {
            auto [r, c] = q.front();
            q.pop();
            if (r == n - 1 && c == n - 1) return true;

            for (const auto& [dr, dc] : dirs) {
                int nr = r + dr, nc = c + dc;
                if (nr < 0 || nr >= n || nc < 0 || nc >= n) continue;
                if (seen[nr][nc] || grid[nr][nc] > t) continue;
                seen[nr][nc] = true;
                q.push({nr, nc});
            }
        }
        return false;
    }
};
