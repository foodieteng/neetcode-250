寫法 A · Union-Find / Kruskal// EDGES BY |Δ|

p1631_uf.cpp C++17 
class Solution {
private:
    vector<int> parent;

    int Find(int x) {
        return parent[x] < 0 ? x : parent[x] = Find(parent[x]);
    }

    void Union(int x, int y) {
        int a = Find(x), b = Find(y);
        if (a == b) return;
        if (-parent[a] < -parent[b]) swap(a, b);
        parent[a] += parent[b];
        parent[b] = a;
    }

public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int m = heights.size(), n = heights[0].size();
        auto id = [&](int r, int c) { return r * n + c; };

        vector<tuple<int, int, int>> edges;        // (|Δ|, u, v)
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (j + 1 < n) {
                    int w = abs(heights[i][j] - heights[i][j + 1]);
                    edges.push_back({w, id(i, j), id(i, j + 1)});
                }
                if (i + 1 < m) {
                    int w = abs(heights[i][j] - heights[i + 1][j]);
                    edges.push_back({w, id(i, j), id(i + 1, j)});
                }
            }
        }
        sort(edges.begin(), edges.end());            // tuple 依第一項 |Δ| 排序

        parent.assign(m * n, -1);
        int src = id(0, 0), dst = id(m - 1, n - 1);
        for (auto& [w, u, v] : edges) {
            Union(u, v);
            if (Find(src) == Find(dst)) return w;     // 連通 → 當前邊權 = 答案
        }
        return 0;
    }
};

p1631_dijkstra.cpp C++17 
class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int m = heights.size(), n = heights[0].size();
        vector<vector<int>> dist(m, vector<int>(n, INT_MAX));
        dist[0][0] = 0;

        const vector<pair<int, int>> dirs = {{-1,0},{1,0},{0,-1},{0,1}};
        using T = tuple<int, int, int>;               // (effort, r, c)
        priority_queue<T, vector<T>, greater<T>> pq;
        pq.push({0, 0, 0});

        while (!pq.empty()) {
            auto [d, r, c] = pq.top();
            pq.pop();
            if (r == m - 1 && c == n - 1) return d;    // 彈出終點 = 答案
            if (d > dist[r][c]) continue;              // 過期條目

            for (auto [dr, dc] : dirs) {
                int nr = r + dr, nc = c + dc;
                if (nr < 0 || nr >= m || nc < 0 || nc >= n) continue;
                int nd = max(d, abs(heights[nr][nc] - heights[r][c]));
                if (nd < dist[nr][nc]) {            // effort = 沿途最大 |Δ|
                    dist[nr][nc] = nd;
                    pq.push({nd, nr, nc});
                }
            }
        }
        return 0;
    }
};

class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        int R = heights.size(), C = heights[0].size();
        int lo = 0, hi = 0;
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (j + 1 < C) hi = max(hi, abs(heights[i][j + 1] - heights[i][j]));
                if (i + 1 < R) hi = max(hi, abs(heights[i + 1][j] - heights[i][j]));
            }
        }

        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            if (reach(heights, mid)) hi = mid;       // 可達 → 縮小上界
            else lo = mid + 1;                       // 不可達 → 抬高下界
        }
        return lo;
    }

private:
    bool reach(const vector<vector<int>>& h, int limit) {
        int R = h.size(), C = h[0].size();
        queue<pair<int, int>> q;
        vector<vector<bool>> seen(R, vector<bool>(C, false));
        q.push({0, 0});
        seen[0][0] = true;

        const vector<pair<int, int>> dirs = {{-1,0},{1,0},{0,-1},{0,1}};
        while (!q.empty()) {
            auto [r, c] = q.front();
            q.pop();
            if (r == R - 1 && c == C - 1) return true;

            for (auto [dr, dc] : dirs) {
                int nr = r + dr, nc = c + dc;
                if (nr < 0 || nr >= R || nc < 0 || nc >= C || seen[nr][nc]) continue;
                if (abs(h[nr][nc] - h[r][c]) > limit) continue;   // 邊權超過門檻,不能跨
                seen[nr][nc] = true;
                q.push({nr, nc});
            }
        }
        return false;
    }
};
