class Solution {
private:
    vector<int> parent;

    int Find(int x) {
        return parent[x] < 0 ? x : parent[x] = Find(parent[x]);
    }

    bool Union(int x, int y) {
        int rx = Find(x);
        int ry = Find(y);
        if (rx == ry) {
            return false;
        }
        if (-parent[rx] < -parent[ry]) {
            swap(rx, ry);
        }
        parent[rx] += parent[ry];
        parent[ry] = rx;
        return true;
    }

    // 建 MST:skip 跳過某邊、force 強制先放某邊;回傳權重或 INT_MAX
    int kruskal(int n, vector<vector<int>>& edges, vector<int>& order,
                int skip, int force) {
        parent.assign(n, -1);
        int total = 0;
        int add = 0;

        if (force != -1) {                 // 先強制放入 force 這條邊
            Union(edges[force][0], edges[force][1]);
            total += edges[force][2];
            add++;
        }

        for (int idx : order) {            // 照權重由小到大掃描
            if (idx == skip) {
                continue;                  // 跳過被 skip 的邊
            }
            if (add >= n - 1) {
                break;                     // 已選滿 n-1 條
            }
            if (Union(edges[idx][0], edges[idx][1])) {
                add++;
                total += edges[idx][2];
            }
        }
        return add == n - 1 ? total : INT_MAX;   // 沒連滿 = 不連通
    }

public:
    vector<vector<int>> findCriticalAndPseudoCriticalEdges(
            int n, vector<vector<int>>& edges) {
        int m = edges.size();

        vector<int> order(m);              // 間接排序:排索引不排邊
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b) {
            return edges[a][2] < edges[b][2];
        });

        int mst = kruskal(n, edges, order, -1, -1);   // 基準 MST 權重

        vector<int> critical;
        vector<int> pseudo;
        for (int i = 0; i < m; i++) {
            if (kruskal(n, edges, order, i, -1) > mst) {
                critical.push_back(i);                 // 拿掉變貴 → 關鍵
            } else if (kruskal(n, edges, order, -1, i) == mst) {
                pseudo.push_back(i);                   // 硬塞仍等 → 偽關鍵
            }
        }
        return {critical, pseudo};
    }
};

複雜度分析// COMPLEXITY

步驟成本說明
間接排序O(m log m)只排一次索引
基準 MSTO(m · α)一次 Kruskal
每邊 skip / forceO(m · α)最多 2 次 Kruskal × m 條邊
Find / Union≈ O(α)路徑壓縮 + 按秩合併
T = O(m² · α)　　S = O(n + m)

m = 邊數,n = 點數。α 為反 Ackermann,實務視為常數。

邊界與陷阱// PITFALLS

陷阱後果正解
直接 sort(edges) 索引被打亂 用間接排序排 order,保留原始邊編號
skip 忘了判「不連通」 漏抓關鍵邊 沒連滿回 INT_MAX,它 > mst 自然算關鍵
先測 pseudo 再測 critical 分類重疊 先 skip 判關鍵,否則才 force 判偽關鍵(互斥)
force 沒把邊「先」放進去 測不到強制效果 force 要先 Union + 加權重 + 計數,再跑其餘
vector<int> order{m} 變成 1 個值 m 要 order(m)(m 個 0)才是大小建構子

// KRUSKAL(skip, force): 一個函式三種用途 skip=-1, force=-1 → 純建 MST(基準 mst) skip=i, force=-1 → 不准用第 i 條邊(關鍵測試) skip=-1, force=i → 先塞第 i 條邊再建(偽關鍵測試) add 計已選邊數;add == n-1 才算連通,否則回 INT_MAX
