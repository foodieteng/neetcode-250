Tarjan 求橋// dfn / low + edge id

p1192.cpp C++17 
class Solution {
    vector<vector<pair<int,int>>> g;   // {鄰居, 邊 id}
    vector<int> dfn, low;
    vector<vector<int>> ans;
    int idx = 0;

    void dfs(int u, int pe) {                // pe = 進來的邊 id,不是父節點!
        dfn[u] = low[u] = idx++;
        for (auto [v, id] : g[u]) {
            if (id == pe) continue;          // 不走回頭那條邊
            if (dfn[v] == -1) {           // 樹邊
                dfs(v, id);
                low[u] = min(low[u], low[v]);
                if (low[v] > dfn[u]) ans.push_back({u, v});  // 橋!
            } else {                        // 回邊
                low[u] = min(low[u], dfn[v]);
            }
        }
    }

public:
    vector<vector<int>> criticalConnections(int n, vector<vector<int>>& connections) {
        g.assign(n, {}); dfn.assign(n, -1); low.assign(n, 0);
        for (int i = 0; i < connections.size(); i++) {
            int a = connections[i][0], b = connections[i][1];
            g[a].push_back({b, i}); g[b].push_back({a, i});  // 同一 id
        }
        for (int i = 0; i < n; i++)
            if (dfn[i] == -1) dfs(i, -1);
        return ans;
    }
};

// 樹邊 vs 回邊:low 更新為什麼不同 樹邊 u→v:v 是新節點,遞迴完後 low[v] 已算好 → low[u] = min(low[u], low[v]) (繼承整個子樹) 回邊 u→v:v 是祖先(已訪),不再遞迴 → low[u] = min(low[u], dfn[v]) (只能跳到 v 的「時間戳」) 用 dfn[v] 而非 low[v]:回邊只保證跳到 v 本身,不保證 v 的 low
// 為什麼邊 id 能解重邊 u—v 有兩條邊 e0, e1(平行) 從 u 走 e0 到 v,pe=e0 v 看鄰居時遇到 e0(skip,回頭)、也遇到 e1 e1 != pe → 當回邊處理 → low[v]=min(.., dfn[u]) → v 繞得回 u → (u,v) 不是橋 ✓ 父節點法會把 e1 也當「回父親」skip → 誤判成橋 ✗
// 範例 trace(三角 + 1-3) dfn = [0,1,2,3] low = [0,0,0,3] (1,2): low[2]=0 > dfn[1]=1 ? 否 → 非橋(2 繞回 0) (1,3): low[3]=3 > dfn[1]=1 ? 是 → 橋 (1,3)
