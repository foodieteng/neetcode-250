寫法 A · DFS 三色後序// DFS POST-ORDER

p269_dfs.cpp C++17 
class Solution {
private:
    unordered_map<char, unordered_set<char>> graph;
    unordered_map<char, int> color;   // 0=白(未訪) 1=灰(訪問中) 2=黑(完成)
    string res;

    // 回傳 false = 偵測到環
    bool dfs(char u) {
        color[u] = 1;                          // 進入:標灰
        for (char v : graph[u]) {
            if (color[v] == 1) {
                return false;                  // 撞到灰 → 有環
            }
            if (color[v] == 0 && !dfs(v)) {
                return false;
            }
        }
        color[u] = 2;                          // 完成:標黑
        res += u;                              // 後序加入(逆拓樸序)
        return true;
    }

public:
    string alienOrder(vector<string>& words) {
        for (auto& w : words) {
            for (char c : w) {
                graph[c];                       // 確保節點存在
                color[c] = 0;
            }
        }

        for (int i = 0; i + 1 < (int)words.size(); i++) {
            string& w1 = words[i];
            string& w2 = words[i + 1];
            int minL = min(w1.size(), w2.size());
            int j = 0;
            for (; j < minL; j++) {
                if (w1[j] == w2[j]) {
                    continue;
                }
                graph[w1[j]].insert(w2[j]);    // set 天生去重
                break;
            }
            if (j == minL && w1.size() > w2.size()) {
                return "";                     // 前綴矛盾
            }
        }

        for (auto& [c, _] : color) {
            if (color[c] == 0 && !dfs(c)) {
                return "";                     // 有環
            }
        }

        reverse(res.begin(), res.end());       // 後序反轉 → 正拓樸序
        return res;
    }
};

寫法 B · Kahn BFS 入度// KAHN BFS

p269_bfs.cpp C++17 
class Solution {
private:
    unordered_map<char, unordered_set<char>> graph;
    unordered_map<char, int> indegree;

public:
    string alienOrder(vector<string>& words) {
        int n = words.size();

        for (auto& w : words) {
            for (char c : w) {
                indegree[c] = 0;
                graph[c] = unordered_set<char>();
            }
        }

        for (int i = 0; i < n - 1; i++) {
            string& w1 = words[i];
            string& w2 = words[i + 1];
            int minL = min(w1.size(), w2.size());
            int j = 0;
            for (; j < minL; j++) {
                if (w1[j] == w2[j]) {
                    continue;
                }
                if (!graph[w1[j]].count(w2[j])) {   // 新邊才加,避免入度重複
                    graph[w1[j]].insert(w2[j]);
                    indegree[w2[j]]++;
                }
                break;
            }
            if (j == minL && w1.size() > w2.size()) {
                return "";                          // 前綴矛盾
            }
        }

        queue<char> q;
        for (auto& [c, deg] : indegree) {
            if (deg == 0) {
                q.push(c);
            }
        }

        string res;
        while (!q.empty()) {
            char u = q.front();
            q.pop();
            res += u;
            for (char v : graph[u]) {
                if (--indegree[v] == 0) {
                    q.push(v);
                }
            }
        }

        return res.size() == indegree.size() ? res : "";   // 沒排完 = 有環
    }
};

兩種寫法對照// A vs B

面向A · DFS 三色B · Kahn BFS
排序方式遞迴後序 + 反轉入度 0 逐層剝除
判環撞到灰色節點res.size() != 字母數
去重set 自動去重即可要 count 檢查才 indegree++
複雜度O(C + V + E)O(C + V + E)
風險遞迴深度(字母 ≤ 26,無虞)忘了 count 會重複計入度
兩者建圖完全相同,只差拓樸骨架。B 的去重很關鍵:同一條邊被多對單字重複推出時,若不檢查就會把 indegree 多加,導致該字母永遠進不了佇列 → 誤判有環。A 用 set 天生免疫這問題。
