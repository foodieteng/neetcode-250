Kosaraju 兩趟 DFS// finish-order → reverse graph

scc_kosaraju.cpp C++17 
#include <bits/stdc++.h>
using namespace std;
int main() {
    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<vector<int>> g(n + 1), rg(n + 1);   // g 原圖 · rg 反圖
    while (m--) {
        int a, b; cin >> a >> b;
        g[a].push_back(b); rg[b].push_back(a);          // a→b 反向 = b→a
    }

    vector<bool> visit(n + 1, false);
    vector<int> comp(n + 1);
    stack<int> order;

    auto dfs1 = [&](auto&& self, int u) {       // 第一趟(原圖):壓完成序
        if (visit[u]) return;
        visit[u] = true;
        for (auto& v : g[u]) self(self, v);
        order.push(u);                             // 後代都做完才壓自己
    };
    auto dfs2 = [&](auto&& self, int u, int k) { // 第二趟(反圖):標同組
        if (comp[u]) return;
        comp[u] = k;
        for (auto& v : rg[u])
            if (!comp[v]) self(self, v, k);
    };

    for (int i = 1; i <= n; i++) dfs1(dfs1, i);       // 建完成序

    int k = 0;
    while (!order.empty()) {
        int u = order.top(); order.pop();          // 由完成最晚的開始
        if (comp[u]) continue;
        k++; dfs2(dfs2, u, k);                       // 一棵樹 = 一個 SCC
    }

    cout << k << endl;
    for (int i = 1; i <= n; i++) cout << comp[i] << " ";
    return 0;
}
