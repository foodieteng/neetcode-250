寫法 A · DFS 後序反轉// POST-ORDER + REVERSE

p210_dfs.cpp C++17 
class Solution {
public:
    vector<int> findOrder(int numCourses, vector<vector<int>>& prerequisites) {
        vector<int> visit(numCourses, 0), order;   // visit:0未訪/1進行中/2完成
        vector<vector<int>> graph(numCourses);

        for (auto& p : prerequisites) {
            int a = p[0], b = p[1];
            graph[b].push_back(a);              // 邊 b→a(先修 b 指向 a)
        }

        for (int i = 0; i < numCourses; i++)
            if (hasCycle(graph, visit, order, i)) return {};   // 有環 → 回空

        reverse(order.begin(), order.end());          // 後序反轉 = 拓撲序
        return order;
    }

    bool hasCycle(vector<vector<int>>& graph, vector<int>& visit, vector<int>& order, int u) {
        if (visit[u] == 1) return true;          // 回到「進行中」的節點 → 環
        if (visit[u] == 2) return false;         // 已完成,不必再走

        visit[u] = 1;                            // 進行中(在當前遞迴路徑上)
        for (int v : graph[u])
            if (hasCycle(graph, visit, order, v)) return true;
        visit[u] = 2;                            // 完成

        order.push_back(u);                       // 後序:所有後代排完才加自己
        return false;
    }
};

寫法 B · BFS / Kahn(入度)// INDEGREE

p210_kahn.cpp C++17 
class Solution {
public:
    vector<int> findOrder(int numCourses, vector<vector<int>>& prerequisites) {
        vector<int> indeg(numCourses, 0);
        vector<vector<int>> graph(numCourses);

        for (auto& p : prerequisites) {
            int b = p[1], a = p[0];
            graph[b].push_back(a);
            indeg[a]++;                          // a 多一個前置
        }

        queue<int> q;
        for (int i = 0; i < numCourses; i++)
            if (indeg[i] == 0) q.push(i);          // 沒前置的先上

        vector<int> order;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            order.push_back(u);
            for (int v : graph[u])
                if (--indeg[v] == 0) q.push(v);      // 前置全上完 → 可上
        }
        return order.size() == numCourses ? order : vector<int>();  // 少了 = 有環
    }
};

Kahn 不需要 sz = q.size() 分層 —— 這題不數層數,直接一個一個 pop 即可(和 994 要數分鐘不同)。若要層數才用 sz。

對照與陷阱// COMPARE + PITFALLS

A · DFS 後序B · BFS / Kahn
核心後代排完才 push,最後反轉入度 0 入隊,逐個剝
判環三態:走到「進行中(1)」order 長度 < numCourses
順序來源後序完成序反轉出隊順序即拓撲序
陷阱後果正解
邊建反(a→b)順序整個顛倒要 b→a(先修指向後修)
DFS 忘了反轉 order拿到反拓撲序後序完成序要 reverse
只用兩態(訪過/沒訪)判環誤判 DAG 有環要三態:1=在當前路徑、2=已完成
Kahn 沒檢查 order 長度有環也回半套順序size() == numCourses 才合法

細節走讀// WALKTHROUGH

// A · 為什麼「後序反轉」= 拓撲序 一個節點只有在「所有後代都 push 進 order」後,自己才 push → 在 order(後序)裡,它排在所有後代「之後」 反轉後 → 它排在後代「之前」= 拓撲序要的「前置在前」 判環三態:0未訪 / 1當前路徑上 / 2完成 走到 1 = 沿當前遞迴路徑繞回自己 → 有環
// B · 為什麼 Kahn 能判環 indeg[x] = x 還有幾個前置沒上;==0 才能上、入隊 上完一門 → 它的後繼各少一個前置(indeg−1),歸零就入隊 環上的節點:彼此互為前置,indeg 永遠 >0 → 進不了佇列 所以有環時 order 收不滿 → size() < numCourses → 判環
// 兩者都對:順序可能不同,都合法 DAG 的拓撲序通常不唯一(有多個合法答案) DFS 反轉給一種、Kahn 給另一種,LeetCode 都接受 挑順手的寫:面試想展示遞迴 → DFS;想直觀、免遞迴 → Kahn
PREV概念與動畫 NEXT複習與 Trace
