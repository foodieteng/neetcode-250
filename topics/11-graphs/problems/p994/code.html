class Solution {
public:
    int orangesRotting(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        queue<pair<int, int>> q;
        int fresh = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0) continue;
                if (grid[i][j] == 1) fresh++;            // 數新鮮總數
                if (grid[i][j] == 2) q.push({i, j});  // 所有腐爛一起入隊(多源)
            }
        }

        if (fresh == 0) return 0;                     // 沒有新鮮 → 0 分鐘

        const vector<pair<int, int>> dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};
        int time = 0;
        while (!q.empty()) {
            int sz = q.size();                       // 這一波(這一分鐘)的所有格
            for (int _ = 0; _ < sz; _++) {
                auto [r, c] = q.front(); q.pop();
                for (auto [dr, dc] : dirs) {
                    int nr = r + dr, nc = c + dc;
                    if (nr < 0 || nr >= m || nc < 0 || nc >= n || grid[nr][nc] != 1) continue;
                    if (--fresh == 0) return time + 1;       // 最後一顆腐爛 → 立刻回這分鐘
                    grid[nr][nc] = 2;                    // 只感染新鮮(== 1)的
                    q.push({nr, nc});
                }
            }
            time++;                                    // 清完一整波 = 過一分鐘
        }
        return -1;                                   // 佇列空但仍有 fresh → 爛不到
    }
};

複雜度分析// COMPLEXITY

T = O(m·n)　　S = O(m·n)

建源掃一遍 O(mn);BFS 每格最多入隊、出隊一次(腐爛時),鄰居檢查常數。總計 O(mn)。空間為佇列 O(mn)。

步驟成本說明
掃描建源 + 數 freshO(mn)找 2 入隊、數 1
逐層 BFS 擴散O(mn)每格進出佇列一次
額外空間O(mn)佇列(腐爛就地寫回 grid)

注意 · 邊界與陷阱// CAUTION · PITFALLS

陷阱後果正解
沒有 sz = q.size() 分批不同分鐘混在一起,分鐘數錯每輪先鎖定這一波大小,一次清完再 time++
忘了數 fresh / 判 −1有孤島也回數字fresh 追蹤新鮮;佇列空仍 >0 → -1
漏掉 fresh==0 早退本來就沒新鮮卻回非 0開頭 if (fresh==0) return 0;
感染條件寫錯(非 != 1)重複感染 / 覆蓋只感染 grid[nr][nc] == 1 的新鮮
把源當一個門、非全部分鐘算多先把所有 2 一起入隊(多源)

// 為什麼要 sz = q.size() 分批 BFS 佇列裡同時混著「這一波」和「下一波剛入隊的」 先記住 sz = 這一波的格數,只處理 sz 個 = 只清「這一分鐘」的腐爛 處理完這 sz 個才 time++,確保「一波剛好對應一分鐘」 若不分批,新入隊的會被當同一分鐘處理 → 分鐘數算少
// --fresh == 0 return time + 1 的 +1 time 是「已經過的完整分鐘數」 現在正在把鄰居從新鮮變腐爛,這動作發生在「下一分鐘」= time+1 所以最後一顆新鮮在此刻腐爛 → 答案是 time+1 (用早退省掉最後一次 time++;等價於清完再 time++ 再回 time)
// 和 286 牆與門的對照 同一個引擎:所有源一起入隊的多源 BFS 286 要「每格到最近門的距離」→ 距離寫回格子,不必數波 994 要「全部腐爛的總分鐘」→ 是 BFS 的層數,要用 sz 數波 記憶點:要「距離值」不用數波;要「幾層/幾分鐘」才要 sz 分批
