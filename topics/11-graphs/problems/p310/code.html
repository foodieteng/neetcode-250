剝葉子找中心// peel leaves layer by layer

p310.cpp C++17 
class Solution {
public:
    vector<int> findMinHeightTrees(int n, vector<vector<int>>& edges) {
        if (n == 1) return {0};          // 特例
        if (n == 2) return {0, 1};

        vector<vector<int>> graph(n);
        vector<int> deg(n, 0);
        for (auto e : edges) {
            int a = e[0], b = e[1];
            graph[a].push_back(b); graph[b].push_back(a);
            deg[a]++; deg[b]++;
        }

        queue<int> q;
        for (int i = 0; i < n; i++)
            if (deg[i] == 1) q.push(i);       // 初始葉子

        while (n > 2) {                    // 剩 > 2 才繼續剝
            int sz = q.size();              // 綁定「這一層」
            n -= sz;
            for (int _ = 0; _ < sz; _++) {
                int u = q.front(); q.pop();
                for (int v : graph[u])
                    if (--deg[v] == 1) q.push(v);   // 降到 1 = 下一層的葉
            }
        }

        vector<int> ans;
        while (!q.empty()) { ans.push_back(q.front()); q.pop(); }  // 剩 1~2 = 中心
        return ans;
    }
};
