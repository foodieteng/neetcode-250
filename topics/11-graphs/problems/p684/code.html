Union-Find 找成環邊// first failing union

p684.cpp C++17 
class Solution {
    vector<int> parent;                        // <0:根,-值=size;>=0:父

    int Find(int x) {
        return parent[x] < 0 ? x : parent[x] = Find(parent[x]);  // 路徑壓縮
    }

    bool Union(int x, int y) {
        int rootX = Find(x), rootY = Find(y);
        if (rootX == rootY) return false;        // 已同組 → 這條邊成環
        if (-parent[rootX] < -parent[rootY]) swap(rootX, rootY);
        parent[rootX] += parent[rootY];
        parent[rootY] = rootX;
        return true;
    }

public:
    vector<int> findRedundantConnection(vector<vector<int>>& edges) {
        int n = edges.size();
        parent.assign(n + 1, -1);              // 節點 1..n(0 不用)

        for (auto e : edges) {
            int u = e[0], v = e[1];
            if (!Union(u, v)) return {u, v};   // 第一條同根的邊 = 答案
        }
        return {};
    }
};

注意 · 原版多打的一行// MY NOTE

原本在 return 之後又寫了一次 Union(u, v);
像這樣: 
if (!Union(u, v)) return {u, v};
Union(u, v);   // ← 這行是多餘的
當 Union(u,v) 第一次成功(回 true)時,if 不觸發、往下走到第二個 Union(u,v) —— 但此時兩端已經同根,這次 Union 只會回 false 並什麼都不做。所以它無害但完全多餘,拿掉更乾淨。一條邊 union 一次就夠。

陷阱// PITFALLS

陷阱後果正解
parent 開 n 而非 n+1節點是 1..n,越界大小 n+1(0 不用)
回傳第一條成功的邊邏輯反了回第一條 失敗(同根)的
先掃完再找環找不到「最後出現」那條邊掃邊判,遇失敗即回
同一條邊 union 兩次多餘(無害)一條一次就好
