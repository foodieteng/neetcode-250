class Solution {
public:
    vector<vector<int>> graph;
    vector<bool> visit;

    int countComponents(int n, vector<vector<int>>& edges) {
        graph.assign(n, {});
        visit.assign(n, false);
        for (auto e : edges) {
            int u = e[0], v = e[1];
            graph[u].push_back(v); graph[v].push_back(u);   // 無向:兩邊都記
        }

        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (visit[i]) continue;
            cnt++;                          // 碰到未訪點 = 新的一塊
            dfs(i);                        // 把整片染訪
        }
        return cnt;
    }

    void dfs(int u) {
        visit[u] = true;
        for (auto v : graph[u]) {
            if (visit[v]) continue;
            dfs(v);
        }
    }
};

寫法 B · Union-Find(數根)// COUNT ROOTS

p323_uf.cpp C++17 
class Solution {
    vector<int> parent;                        // <0:根,-值=size;>=0:父

    int Find(int x) {
        return parent[x] < 0 ? x : parent[x] = Find(parent[x]);  // 路徑壓縮
    }

    void Union(int x, int y) {
        int rootX = Find(x), rootY = Find(y);
        if (rootX == rootY) return;            // 已同組,不動
        if (-parent[rootX] < -parent[rootY]) swap(rootX, rootY);
        parent[rootX] += parent[rootY];        // 按 size 合併
        parent[rootY] = rootX;
    }

public:
    int countComponents(int n, vector<vector<int>>& edges) {
        parent.resize(n, -1);
        for (auto &e : edges) Union(e[0], e[1]);

        int cnt = 0;
        for (int i = 0; i < n; i++)
            if (parent[i] < 0) cnt++;          // 每個根 = 一塊
        return cnt;
    }
};

和 261 的並查集一模一樣,只是 Union 不必回傳 bool、最後改成數根。也可用 cnt = n − 成功合併次數。
