class Solution {
public:
    vector<bool> checkIfPrerequisite(int numCourses,
            vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<vector<bool>> table(numCourses, vector<bool>(numCourses, false));
        vector<vector<int>> graph(numCourses);
        vector<int> indeg(numCourses, 0);

        for (auto& pre : prerequisites) {
            int a = pre[0], b = pre[1];
            graph[a].push_back(b);              // 邊 a→b(a 是 b 的先修)
            indeg[b]++;
        }

        queue<int> q;
        for (int i = 0; i < numCourses; i++)
            if (indeg[i] == 0) q.push(i);

        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : graph[u]) {
                table[u][v] = true;                // 直接:u 是 v 的先修
                if (--indeg[v] == 0) q.push(v);

                for (int i = 0; i < numCourses; i++) {
                    if (i == u || !table[i][u]) continue;
                    table[i][v] = true;            // 傳遞:i 是 u 的先修 → 也是 v 的
                }
            }
        }

        vector<bool> ans;
        for (auto& qy : queries) ans.push_back(table[qy[0]][qy[1]]);   // O(1) 查表
        return ans;
    }
};

class Solution {
public:
    vector<bool> checkIfPrerequisite(int n,
            vector<vector<int>>& prerequisites, vector<vector<int>>& queries) {
        vector<vector<bool>> reach(n, vector<bool>(n, false));

        for (auto& p : prerequisites)
            reach[p[0]][p[1]] = true;            // 直接邊 a→b(a 是 b 的先修)

        for (int k = 0; k < n; k++)              // 中繼點
            for (int i = 0; i < n; i++)
                for (int j = 0; j < n; j++)
                    if (reach[i][k] && reach[k][j])   // i→k 且 k→j
                        reach[i][j] = true;            // ⟹ i→j

        vector<bool> ans;
        for (auto& q : queries) ans.push_back(reach[q[0]][q[1]]);   // O(1) 查表
        return ans;
    }
};

只有 三重迴圈 + 一個布林矩陣,不必建圖、不必拓撲序。每個 k 都當一次「中繼站」,把所有經過 k 的間接可達補上。順序無關 —— 最外層跑完每個 k 就收斂。本題 n ≤ 100,O(n³)=10⁶ 綽綽有餘,是面試最快寫、最不易錯的版本。
