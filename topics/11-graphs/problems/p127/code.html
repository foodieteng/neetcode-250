class Solution {
public:
    int ladderLength(string beginWord, string endWord, vector<string>& wordList) {
        unordered_set<string> dict(wordList.begin(), wordList.end()), seen;
        int step = 1;
        queue<string> q;
        seen.insert(beginWord);
        q.push(beginWord);

        while (!q.empty()) {
            int sz = q.size();                // 綁定這一層
            for (int _ = 0; _ < sz; _++) {
                string w = q.front(); q.pop();

                for (int i = 0; i < w.size(); i++) {
                    string cache = w;                // 記住原字,等下還原
                    for (int j = 0; j < 26; j++) {
                        w[i] = j + 'a';            // 第 i 位換成 a..z
                        if (!dict.count(w) || seen.count(w)) continue;
                        if (w == endWord) return step + 1;  // 命中 → 下一層長度

                        seen.insert(w);
                        q.push(w);
                    }
                    w = cache;                       // 還原,換下一個位置
                }
            }
            step++;                             // 整層做完,層數 +1
        }
        return 0;                             // 到不了
    }
};

關鍵行// KEY LINES

行意義
dict(...) 建 hashsetO(1) 查「這個變體是不是有效字」
string cache = w換字母前存原字,試完還原
w[i] = j + 'a'第 i 位試 26 個字母(即時生鄰居)
!dict.count || seen.count不在字典 / 已訪 → 跳過
w == endWord → step+1命中終點,回下一層長度
step++(層外)整層擴散完才 +1 = BFS 分層

陷阱// PITFALLS

陷阱後果正解
換字母後沒還原同一個字被改爛,鄰居生錯cache 存原字、內圈結束還原
沒用 sz 綁層step 亂加,答案錯sz = q.size() 一次一層
入隊時不標 seen同字重複入隊、可能爆量入隊當下就 seen.insert
endWord 不在字典永遠命中不了字典裡沒有就直接回 0(自然發生)
回 step 而非 step+1少算一步命中終點時當前 step 是「上一層」,+1
