完整實作 · 反向淹沒// FLOOD FROM OCEANS

class Solution {
public:
    int m, n;
    const vector<pair<int, int>> dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

    vector<vector<int>> pacificAtlantic(vector<vector<int>>& heights) {
        m = heights.size(); n = heights[0].size();
        vector<vector<bool>> p_set(m, vector<bool>(n, false)), a_set(m, vector<bool>(n, false));

        for (int i = 0; i < m; i++) {             // 左緣→太平洋、右緣→大西洋
            dfs(heights, p_set, i, 0);
            dfs(heights, a_set, i, n - 1);
        }
        for (int i = 0; i < n; i++) {             // 上緣→太平洋、下緣→大西洋
            dfs(heights, p_set, 0, i);
            dfs(heights, a_set, m - 1, i);
        }

        vector<vector<int>> ans;
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (p_set[i][j] && a_set[i][j])       // 兩海都可達 = 答案
                    ans.push_back({i, j});
            }
        }
        return ans;
    }

    void dfs(vector<vector<int>>& heights, vector<vector<bool>>& seen, int r, int c) {
        seen[r][c] = true;                     // 進入即標記
        for (auto [dr, dc] : dirs) {
            int nr = r + dr, nc = c + dc;
            if (nr < 0 || nr >= m || nc < 0 || nc >= n) continue;
            if (seen[nr][nc] || heights[nr][nc] < heights[r][c]) continue;  // 只往「≥ 當前」的鄰居逆流
            dfs(heights, seen, nr, nc);
        }
    }
};

複雜度分析// COMPLEXITY

T = O(m·n)　　S = O(m·n)

兩次淹沒(太平洋、大西洋),每格在每次 DFS 最多被訪一次(seen 去重);最後掃一遍取交集 O(mn)。空間為兩個布林網格 + 遞迴堆疊。

步驟成本說明
太平洋淹沒O(mn)每格最多訪一次
大西洋淹沒O(mn)另一個 seen
掃描取交集O(mn)p_set && a_set

注意 · 邊界與陷阱// CAUTION · PITFALLS

陷阱後果正解
淹沒條件寫成 > 或 <=方向反、答案全錯反向逆流:只走 heights[鄰] ≥ 當前(即 < 才 continue)
兩個海共用一個 seen交集失義各自 p_set、a_set 兩個獨立網格
邊界起點搞混海的歸屬錯太平洋=上緣+左緣;大西洋=下緣+右緣
忘了 seen 去重重複遞迴 / 死圈進 dfs 即 seen[r][c]=true,已訪就跳過
先讀 heights[nr][nc] 才驗界越界 · UB先驗界(|| 短路)再比高度

細節走讀// WALKTHROUGH

// 為什麼淹沒條件是「≥」(逆流向上) 正向:水從 A 流到 B 要 height[B] ≤ height[A](往低處) 反向:從海邊出發、把邊反過來走 → height[鄰] ≥ height[當前] 才能上去 「從海邊逆流能到 X」⟺「X 能順流到海」,兩者等價 所以 dfs 的推進條件是 heights[nr][nc] >= heights[r][c] (程式寫成 < 就 continue,等同於「>= 才走」)
// 起點:哪些邊屬哪個海 太平洋(p_set)= 上緣(row 0 每格)+ 左緣(col 0 每格) 大西洋(a_set)= 下緣(row m-1 每格)+ 右緣(col n-1 每格) 第一個 for i∈[0,m):dfs(p,i,0) 左緣、dfs(a,i,n-1) 右緣 第二個 for i∈[0,n):dfs(p,0,i) 上緣、dfs(a,m-1,i) 下緣
// 取交集 = 答案 p_set[i][j] = 這格的水能流到太平洋 a_set[i][j] = 能流到大西洋 兩個都 true → 能同時流到兩海 → 加進 ans 最後掃全網格一次收集,順序即 row-major
PREV概念與動畫 NEXT複習與 Trace
// KEY LESSON
從兩海邊界往「≥ 當前」的鄰居逆流 DFS(各一個 seen),取交集即答案;O(mn)。
