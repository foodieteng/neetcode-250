Union-Find over emails// map → union → group

class Solution {
    vector<int> parent;
    int Find(int x) { return parent[x] < 0 ? x : parent[x] = Find(parent[x]); }
    bool Union(int x, int y) {
        int rx = Find(x), ry = Find(y);
        if (rx == ry) return false;
        if (-parent[rx] < -parent[ry]) swap(rx, ry);
        parent[rx] += parent[ry]; parent[ry] = rx; return true;
    }
public:
    vector<vector<string>> accountsMerge(vector<vector<string>>& accounts) {
        unordered_map<string, int> id;
        vector<string> email, owner;          // id → email · id → 名字

        for (const auto& acc : accounts)       // ① 每個 email 給一個 id
            for (int i = 1; i < acc.size(); i++)
                if (!id.count(acc[i])) {
                    id[acc[i]] = email.size();
                    email.push_back(acc[i]);
                    owner.push_back(acc[0]);     // 連單獨 email 也註冊
                }

        parent.assign(email.size(), -1);

        for (const auto& acc : accounts)       // ② 星形 union(不是團)
            for (int i = 2; i < acc.size(); i++)
                Union(id[acc[1]], id[acc[i]]);  // acc[1] 連其餘

        unordered_map<int, vector<string>> buf;
        for (int i = 0; i < (int)email.size(); i++)
            buf[Find(i)].push_back(email[i]);       // ③ 依 root 分組

        vector<vector<string>> ans;
        for (auto& [root, group] : buf) {
            sort(group.begin(), group.end());          // email 字典序
            group.insert(group.begin(), owner[root]);  // 前面掛名字
            ans.push_back(group);
        }
        return ans;
    }
};
