寫法 A · DFS + hashmap// MAIN

p133_hashmap.cpp C++17 
// Node: int val; vector<Node*> neighbors;
class Solution {
public:
    Node* cloneGraph(Node* node) {
        if (!node) return nullptr;                    // 空圖
        unordered_map<Node*, Node*> clone;          // 原節點 → 新節點
        return dfs(node, clone);
    }

    Node* dfs(Node* node, unordered_map<Node*, Node*>& clone) {
        if (clone.count(node)) return clone[node];       // 已複製 → 回傳既有複本

        Node* copy = new Node(node->val);
        clone[node] = copy;                            // 先存 map,再遞迴鄰居(切斷環)
        for (Node* nei : node->neighbors)
            copy->neighbors.push_back(dfs(nei, clone));
        return copy;
    }
};

寫法 B · BFS + hashmap// ITERATIVE

p133_bfs.cpp C++17 
class Solution {
public:
    Node* cloneGraph(Node* node) {
        if (!node) return nullptr;
        unordered_map<Node*, Node*> clone;
        clone[node] = new Node(node->val);            // 先建起點的複本

        queue<Node*> q;
        q.push(node);
        while (!q.empty()) {
            Node* cur = q.front(); q.pop();
            for (Node* nei : cur->neighbors) {
                if (!clone.count(nei)) {                // 沒複製過 → 新建 + 入隊
                    clone[nei] = new Node(nei->val);
                    q.push(nei);
                }
                clone[cur]->neighbors.push_back(clone[nei]);   // 接上複本的邊
            }
        }
        return clone[node];
    }
};

和 DFS 同樣用 map 去重、切斷環,只是把遞迴換成佇列。要點:出隊 cur 前它的複本必須已存在 —— 所以起點在迴圈外先建、其餘節點都在「第一次當鄰居被看到」時新建並入隊。空間仍 O(V)(map + 佇列)。

寫法 C · Follow-up · map 用 val 索引 → O(1)// O(1) MAP

p133_array.cpp C++17 
class Solution {
public:
    Node* cloneGraph(Node* node) {
        if (!node) return nullptr;
        vector<Node*> clone(101, nullptr);            // 固定 101 格,以 val 為索引 → O(1)
        return dfs(node, clone);
    }

    Node* dfs(Node* node, vector<Node*>& clone) {
        if (clone[node->val]) return clone[node->val];   // val 唯一 → 當索引

        Node* copy = new Node(node->val);
        clone[node->val] = copy;
        for (Node* nei : node->neighbors)
            copy->neighbors.push_back(dfs(nei, clone));
        return copy;
    }
};

只有 map 的型別不同:unordered_map<Node*,Node*>(O(V))→ 固定 vector<Node*>(101)(O(1))。演算法完全一樣。走訪的遞迴堆疊仍 O(V)。

注意 · 對照與陷阱// CAUTION · PITFALLS

陷阱後果正解
先遞迴鄰居、後存 map環回來時還沒存 → 死圈先 clone[node]=copy 再遞迴鄰居
新節點鄰居指向原圖假拷貝、非獨立鄰居也要放 dfs(nei) 的新節點
忘了 node == nullptr空圖崩潰開頭 if (!node) return nullptr;
B:陣列開 100 而非 101val=100 越界val 上限 100 → 要 101 格(index 0..100)
誤以為能做到「真 O(1) 空間」觀念錯圖走訪必須 O(V);O(1) 只指 map 結構

細節走讀// WALKTHROUGH

// 為什麼「先存 map,再遞迴鄰居」 圖有環:A 的鄰居是 B,B 的鄰居又是 A 若先遞迴 B、還沒把 A 存進 map: 複製 B 時要複製它的鄰居 A → A 不在 map → 又新建一個 A → 無限迴圈 正解:一進 dfs、建好 copy 就「立刻」clone[node]=copy 之後 B 遞迴回來看到 A 已在 map,直接取用 → 環被切斷
// DFS vs BFS:同一件事,不同走訪 DFS(遞迴):進 dfs 先存 map、再往鄰居遞迴;靠呼叫堆疊記進度 BFS(佇列):起點先建複本入隊;出隊 cur、掃鄰居,新的就建+入隊,再接邊 兩者都用 map「查得到就重用、查不到就新建」→ 同樣去重、切斷環 差別只在走訪順序(深度 vs 廣度);超深的圖用 BFS 可避免遞迴爆堆疊
// hashmap 版 vs val 陣列版(follow-up)差在哪 hashmap：unordered_map<Node*, Node*> 以「原節點指標」為鍵,大小 O(V) val 陣列：vector<Node*>(101) 以「val(1..100)」為索引,大小 O(1) 兩者邏輯一模一樣,只有 map 的型別不同(DFS、BFS 都可換)
// 注意 · 為什麼不能做到真 O(1) 空間 複製圖必須「走訪」所有節點,而走訪要記錄訪過誰 → 遞迴堆疊 / BFS 佇列 / visited,至少 O(V),省不掉 對照 138(含隨機指標的鏈結串列):有 next 可免費列舉節點 才能用「穿插複本」做到真 O(1);圖沒有 next,做不到 所以這題 follow-up 的「O(1)」只針對 map 結構(換成 val 陣列)
PREV概念與動畫 NEXT複習與 Trace
// KEY LESSON
先存 map 再遞迴鄰居(切斷環);map 可用 val 索引壓到 O(1),但走訪 O(V) 省不掉。
