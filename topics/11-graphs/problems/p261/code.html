Union-Find 判樹// path compression + union by size

p261.cpp C++17 
class Solution {
    vector<int> parent;                        // <0:自己是根,-值=size;>=0:父節點

    int Find(int x) {
        return parent[x] < 0 ? x : parent[x] = Find(parent[x]);  // 路徑壓縮
    }

    bool Union(int x, int y) {
        int rootX = Find(x), rootY = Find(y);
        if (rootX == rootY) return false;        // 已同根 → 這條邊造成環

        if (-parent[rootX] < -parent[rootY])   // 讓 rootX 是較大的樹
            swap(rootX, rootY);
        parent[rootX] += parent[rootY];        // size 相加(兩個負數相加)
        parent[rootY] = rootX;              // 小樹掛到大樹下
        return true;
    }

public:
    bool validTree(int n, vector<vector<int>>& edges) {
        if (edges.size() != n - 1) return false;   // 樹必有 n-1 條邊

        parent.assign(n, -1);                   // n 個獨立點,各為根、size 1
        int add = 0;

        for (auto e : edges) {
            int u = e[0], v = e[1];
            if (Union(u, v)) add++;             // 成功合併次數
        }
        return add == n - 1;                    // n-1 次成功 = 無環且連通
    }
};

關鍵行// KEY LINES

行意義
parent[x] = Find(parent[x])路徑壓縮:沿途節點直接改指向根
if (rootX == rootY) return false兩端已同組 → 這條邊成環
-parent[rootX] < -parent[rootY]比 size,確保大樹當主幹
parent[rootX] += parent[rootY]兩負數相加 = size 合併
add == n - 1成功合併 n−1 次 ⟺ 合法樹

其實 edges.size() != n−1 的前置檢查 + add == n−1 有點重疊:既然邊數已是 n−1,只要沒有任何一條邊成環(add 也會是 n−1)就是樹。留著前置檢查能提早剪枝、語意更清楚。

陷阱// PITFALLS

陷阱後果正解
只檢查連通、沒查邊數有環也可能連通 → 誤判成樹兩條件都要:n−1 邊 + 無環
只檢查 n−1 邊、沒判環一邊成環、一邊斷開也 n−1 條用 union 判環
Find 沒路徑壓縮退化成鏈,單次可能 O(n)parent[x]=Find(...)
Union 不按 size 亂接樹可能長高、變慢小樹掛大樹
忘了 rootX==rootY 就合併自我合併、size 記錯同根先 return false
