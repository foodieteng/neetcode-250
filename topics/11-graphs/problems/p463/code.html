完整實作 · DFS// STEP OFF ISLAND = +1

class Solution {
private:
    int m, n;
    const vector<pair<int, int>> dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

public:
    int islandPerimeter(vector<vector<int>>& grid) {
        m = grid.size(); n = grid[0].size();
        int res = 0;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] != 1) continue;
                dfs(grid, i, j, res);       // 島連通,一次走遍全島
            }
        }
        return res;
    }

    void dfs(vector<vector<int>>& grid, int r, int c, int& res) {
        if (r < 0 || r >= m || c < 0 || c >= n || grid[r][c] == 0) {
            res += 1; return;              // 踩出界 / 踩到水 = 一條周長邊
        }
        if (grid[r][c] == -1) return;       // 已訪陸地,不重複

        grid[r][c] = -1;                   // 標記為已訪

        for (auto [dr, dc] : dirs) {
            int nr = r + dr, nc = c + dc;
            dfs(grid, nr, nc, res);          // 照樣往鄰居走,判斷在 dfs 開頭
        }
    }
};

複雜度分析// COMPLEXITY

T = O(m·n)　　S = O(m·n)

每個陸地格只標記一次(-1),它的四個方向各觸發一次 dfs;非陸地的呼叫在 base case 立刻返回。總呼叫量 O(mn)。空間為遞迴堆疊深度。

步驟成本說明
雙迴圈找起點O(mn)掃到第一個陸地格
DFS 走遍島O(mn)每格標記一次 + 四向遞迴
遞迴堆疊O(mn)最壞(全島蛇形)深度

錯誤筆記 · 判斷放哪裡// PRE-GUARD vs BASE-CASE

// ✗ 我一開始的寫法(判斷寫在呼叫前) for (auto [dr, dc] : dirs) { int nr = r + dr, nc = c + dc; if (nr < 0 || nr >= m || nc < 0 || nc >= n || grid[nr][nc] == 0) continue; ← 直接跳過,沒有 +1 dfs(grid, nr, nc, res); ← 只在對面是陸地時才遞迴 } 問題:朝向水/出界的那條邊 從來沒被數到 → 周長偏小
// ✓ 正確寫法(判斷當 base case) // dfs 開頭: if (出界 || grid[r][c] == 0) { res += 1; return; } ← 踩出島才 +1 if (grid[r][c] == -1) return; ← 已訪 grid[r][c] = -1; for (auto [dr, dc] : dirs) dfs(grid, r+dr, c+dc, res); ← 照樣遞迴,讓「踩出島」真的發生
// 為什麼差在這 周長邊 = 「陸地 → 水/出界」的那一步 pre-guard 把這一步攔在遞迴之外 → 事件沒發生 → 數不到 base-case 讓這一步真的走進 dfs → res += 1 被執行 口訣:要數的東西,必須讓它「被走到」,不能提前 continue 掉。
