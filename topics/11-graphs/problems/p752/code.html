class Solution {
public:
    int openLock(vector<string>& deadends, string target) {
        unordered_set<string> dead(deadends.begin(), deadends.end());
        if (dead.count("0000")) return -1;          // 起點就死 → 開不了
        if (target == "0000") return 0;            // 起點就是目標

        unordered_set<string> seen;
        queue<string> q;
        seen.insert("0000"); q.push("0000");
        int step = 0;

        while (!q.empty()) {
            int sz = q.size();                       // 這一層(轉這一次)的所有狀態
            for (int _ = 0; _ < sz; _++) {
                auto u = q.front(); q.pop();
                if (u == target) return step;          // 第一次出隊 = 最短

                for (int i = 0; i < 4; i++) {
                    string cache = u;

                    u[i] = (cache[i] == '9') ? '0' : cache[i] + 1;  // 上轉:ASCII 直接 +1
                    if (!dead.count(u) && !seen.count(u)) { seen.insert(u); q.push(u); }

                    u[i] = (cache[i] == '0') ? '9' : cache[i] - 1;  // 下轉:ASCII 直接 −1
                    if (!dead.count(u) && !seen.count(u)) { seen.insert(u); q.push(u); }

                    u = cache;                          // ★ 還原,再換下一個輪
                }
            }
            step++;                                    // 清完一層 = 轉了一次
        }
        return -1;                                   // 走不到 target
    }
};

複雜度分析// COMPLEXITY

T = O(10⁴)　　S = O(10⁴)

狀態至多 10⁴ 個,每個產生 4×2=8 個鄰居、每個鄰居 O(4) 比較與雜湊。常數倍,整體 O(10⁴)。空間為 seen + 佇列。

步驟成本說明
建 dead 集合O(D)D = deadends 數

額外空間O(10⁴)seen + queue
