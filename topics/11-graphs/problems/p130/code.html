class Solution {
public:
    int m, n;
    const vector<pair<int, int>> dirs = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};

    void solve(vector<vector<char>>& board) {
        m = board.size(); n = board[0].size();

        for (int r = 0; r < m; r++) {             // 左緣、右緣的 O
            if (board[r][0] == 'O') dfs(board, r, 0);
            if (board[r][n - 1] == 'O') dfs(board, r, n - 1);
        }
        for (int c = 0; c < n; c++) {             // 上緣、下緣的 O
            if (board[0][c] == 'O') dfs(board, 0, c);
            if (board[m - 1][c] == 'O') dfs(board, m - 1, c);
        }

        for (int r = 0; r < m; r++) {             // 翻面
            for (int c = 0; c < n; c++) {
                board[r][c] = (board[r][c] == '#') ? 'O' : 'X';  // # 保留 O,其餘捕獲
            }
        }
    }

    void dfs(vector<vector<char>>& board, int r, int c) {
        board[r][c] = '#';                      // 標記安全(進入即標)
        for (auto [dr, dc] : dirs) {
            int nr = r + dr, nc = c + dc;
            if (nr < 0 || nr >= m || nc < 0 || nc >= n || board[nr][nc] != 'O') continue;  // 只走還是 O 的
            dfs(board, nr, nc);
        }
    }
};

複雜度分析// COMPLEXITY

T = O(m·n)　　S = O(m·n)

邊界 DFS 每格最多訪一次('#' 去重),翻面掃一遍。總造訪 O(mn)。空間為遞迴堆疊,最壞(整盤相連 O)可達 O(mn)。

步驟成本說明
掃邊界 + DFS 標安全O(mn)每格最多訪一次
翻面O(mn)# → O、其餘 → X
遞迴堆疊O(mn)最壞深度

注意 · 邊界與陷阱// CAUTION · PITFALLS

陷阱後果正解
char 忘了引號(寫 == O)編不過 / 型別錯board 是 char → 'O'、'X'、'#' 帶單引號
正著找被包圍、直接翻邊界連通的也被誤翻反過來:先標安全,剩下的才捕獲
DFS 條件沒限 == 'O'重走 # / 撞 X只往還是 'O'的鄰居(#已訪、X牆都跳)
翻面時 '#' 沒還原安全區留下 #'#' → 'O';非 # 一律 'X'
只掃部分邊界漏標某些安全區左、右、上、下四條邊都要當起點
