class Solution {
public:
    vector<int> sortArray(vector<int>& nums) {
        radixSort(nums);          // 本題實際跑的:基數排序(O(n))
        // 也可換成:bubbleSort / selectionSort / insertionSort
        //          quickSort(nums,0,n-1) / mergeSort(nums,0,n-1)
        //          heapSort / shellSort / bucketSort
        return nums;
    }
    // ⋯ 九種排序的實作見下方各節 ⋯
};

簡單排序 · O(n²)// BUBBLE / SELECTION / INSERTION

// 氣泡:相鄰比較換,swapped 旗標可提早結束(近乎有序時 O(n)) void bubbleSort(vector<int>& a) { int n = a.size(); for (int i = 0; i < n - 1; i++) { bool swapped = false; for (int j = 0; j < n - i - 1; j++) if (a[j] > a[j + 1]) { swap(a[j], a[j + 1]); swapped = true; } if (!swapped) break; // 這趟沒換 → 已排好 } } 穩定 · O(1) 空間 · 教學價值高、實務極慢
// 選擇:每趟選未排序區最小,換到最前(交換次數少但不穩定) void selectionSort(vector<int>& a) { int n = a.size(); for (int i = 0; i < n - 1; i++) { int minIdx = i; for (int j = i + 1; j < n; j++) if (a[j] < a[minIdx]) minIdx = j; swap(a[minIdx], a[i]); // 遠距 swap → 可能破壞相等元素順序 } } 不穩定 · O(1) 空間 · 比較數固定 n(n-1)/2,不看輸入
// 插入:維持左側已排序前綴,新 key 插進去(整理手牌) void insertionSort(vector<int>& a) { int n = a.size(); for (int i = 1; i < n; i++) { int key = a[i], j = i - 1; while (j >= 0 && a[j] > key) { a[j + 1] = a[j]; j--; } // 大的右移 a[j + 1] = key; } } 穩定 · O(1) 空間 · 近乎有序時 O(n),小陣列首選

進階比較排序 · O(n log n)// SHELL / QUICK / MERGE / HEAP

// 希爾:間隔 gap 的插入排序,gap 由 n/2 縮到 1 void shellSort(vector<int>& a) { int n = a.size(); for (int gap = n / 2; gap > 0; gap /= 2) for (int i = gap; i < n; i++) { int key = a[i], j = i - gap; while (j >= 0 && a[j] > key) { a[j + gap] = a[j]; j -= gap; } a[j + gap] = key; } } 不穩定 · O(1) 空間 · 複雜度看 gap 序列(~O(n^1.3))
// 快排:Lomuto partition(取最右為 pivot),遞迴左右 void quickSort(vector<int>& a, int l, int r) { if (l >= r) return; int p = partition(a, l, r); quickSort(a, l, p - 1); quickSort(a, p + 1, r); } int partition(vector<int>& a, int l, int r) { int pivot = a[r], i = l; for (int j = l; j < r; j++) if (a[j] <= pivot) { swap(a[j], a[i]); i++; } // ≤pivot 累到左區 swap(a[i], a[r]); // pivot 歸位 return i; } 不穩定 · O(log n) 遞迴堆疊 · 平均 O(n log n)、最壞 O(n²)(隨機 pivot 可避)
// 合併:分治到剩 1,再雙指針合併兩個已排序半段 void mergeSort(vector<int>& a, int l, int r) { if (l >= r) return; int m = l + (r - l) / 2; mergeSort(a, l, m); mergeSort(a, m + 1, r); merge(a, l, m, r); } void merge(vector<int>& a, int l, int m, int r) { vector<int> tmp; int i = l, j = m + 1; while (i <= m && j <= r) tmp.push_back(a[i] <= a[j] ? a[i++] : a[j++]); // ≤ 取左 → 穩定 while (i <= m) tmp.push_back(a[i++]); while (j <= r) tmp.push_back(a[j++]); for (int k = 0; k < (int)tmp.size(); k++) a[l + k] = tmp[k]; } 穩定 · O(n) 額外空間 · 最壞仍 O(n log n),外部排序基礎
// 堆積:建 max-heap,反覆把根換到尾、堆縮小、sift down void heapify(vector<int>& a, int n, int i) { int largest = i, l = 2*i+1, r = 2*i+2; if (l < n && a[l] > a[largest]) largest = l; if (r < n && a[r] > a[largest]) largest = r; if (largest != i) { swap(a[largest], a[i]); heapify(a, n, largest); } } void heapSort(vector<int>& a) { int n = a.size(); for (int i = n/2 - 1; i >= 0; i--) heapify(a, n, i); // build max-heap for (int i = n - 1; i > 0; i--) { swap(a[0], a[i]); // 最大 → 尾 heapify(a, i, 0); // 對縮小的堆 sift } } 不穩定 · O(1) 原地 · 最壞保證 O(n log n),優先佇列的基礎

非比較排序 · O(n)// BUCKET / RADIX

// 桶:依值域散進 n 個桶,桶內排序,再串接 void bucketSort(vector<int>& a) { int n = a.size(); if (n == 0) return; int minv = *min_element(a.begin(), a.end()); int maxv = *max_element(a.begin(), a.end()); double range = double(maxv - minv + 1); vector<vector<int>> buckets(n); for (int x : a) { int idx = (int)(((double)(x - minv) / range) * (n - 1)); // 值 → 桶號 buckets[idx].push_back(x); } a.clear(); for (auto& b : buckets) { sort(b.begin(), b.end()); for (int x : b) a.push_back(x); } } 均勻分布時近 O(n) · O(n) 空間 · 分布不均會退化
// 基數:LSD 逐位穩定計數排序;負數拆出取絕對值、再反轉取負 vector<int> radix(vector<int>& a, int exp) { // 對「(a/exp)%10」這一位做計數排序 int n = a.size(); vector<int> cnt(10, 0), output(n); for (int i = 0; i < n; i++) cnt[(a[i]/exp)%10]++; for (int i = 1; i < 10; i++) cnt[i] += cnt[i-1]; // 前綴和 = 每位的結束位置 for (int i = n - 1; i >= 0; i--) { // 由後往前 → 穩定 int d = (a[i]/exp)%10; output[--cnt[d]] = a[i]; } return output; } void radixSort(vector<int>& a) { vector<int> pos, neg; for (int x : a) (x >= 0 ? pos : neg).push_back(x >= 0 ? x : -x); // 負數存絕對值 int maxp = 0, minp = 0; for (int x : pos) maxp = max(maxp, x); for (int x : neg) minp = max(minp, x); for (int exp = 1; exp <= maxp; exp *= 10) pos = radix(pos, exp); // 正:小→大 for (int exp = 1; exp <= minp; exp *= 10) neg = radix(neg, exp); // 負(絕對值)小→大 reverse(neg.begin(), neg.end()); // 絕對值大 → 最負,放最前 a.clear(); for (int x : neg) a.push_back(-x); // 取負接回 for (int x : pos) a.push_back(x); } 穩定 · O(d·n) ≈ O(n)(位數 d 常數)· 只適用整數/定長鍵
基數的兩個關鍵:(1) 逐位必須穩定 —— 計數排序「由後往前 + 前綴和」保證同一位相同的數維持先前順序,低位的排序結果才不會被高位打亂; (2) 負數處理 —— 基數只認非負,所以把負數取絕對值單獨排,排好後反轉(絕對值大的最負)再取負,接到正數前面。
