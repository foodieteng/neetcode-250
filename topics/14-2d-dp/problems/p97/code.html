寫法 A · 一維滾動(你的版本)// O(s2) — FOLLOW-UP

p97.cpp C++17 
class Solution {
public:
    bool isInterleave(string s1, string s2, string s3) {
        int m = s1.size(), n = s2.size();

        if (m + n != s3.size()) return false;   // 字數對不上 → 免談

        vector<bool> dp(n + 1, false);
        dp[0] = true;

        for (int i = 1; i <= n; i++)              // 第一列:只用 s2
            dp[i] = (s2[i - 1] == s3[i - 1] && dp[i - 1]);

        for (int i = 1; i <= m; i++) {
            dp[0] = dp[0] && s1[i - 1] == s3[i - 1];  // 第一行:只用 s1

            for (int j = 1; j <= n; j++) {
                dp[j] = (dp[j] && s1[i - 1] == s3[i + j - 1])   // dp[j] 舊值 = 上 → 拿 s1
                    || (dp[j - 1] && s2[j - 1] == s3[i + j - 1]); // dp[j-1] 新值 = 左 → 拿 s2
            }
        }

        return dp[n];
    }
};

這正是 follow-up 要的 O(s2.length) 空間。訣竅同 62:更新前的 dp[j] 還是上一列的值(= 「上」= 拿 s1),dp[j−1] 是本列已更新的左鄰(= 「左」= 拿 s2)。因為只依賴上/左、沒有斜對角,壓維不需要暫存變數。dp[0] 每列開頭要單獨更新(第一行只能一路拿 s1)。

寫法 B · 二維(先寫這個最好懂)// 2-D

p97_2d.cpp C++17 
class Solution {
public:
    bool isInterleave(string s1, string s2, string s3) {
        int m = s1.size(), n = s2.size();
        if (m + n != s3.size()) return false;

        vector<vector<bool>> dp(m + 1, vector<bool>(n + 1, false));
        dp[0][0] = true;

        for (int i = 1; i <= m; i++)            // 只用 s1
            dp[i][0] = (s1[i - 1] == s3[i - 1] && dp[i - 1][0]);

        for (int j = 1; j <= n; j++)            // 只用 s2
            dp[0][j] = (s2[j - 1] == s3[j - 1]) && dp[0][j - 1];

        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = (s1[i - 1] == s3[i + j - 1] && dp[i - 1][j])  // 拿 s1(上)
                    || (s2[j - 1] == s3[i + j - 1] && dp[i][j - 1]); // 拿 s2(左)
            }
        }

        return dp[m][n];
    }
};

面試建議先寫這個:兩種來源、兩個邊界一目了然,不必想「dp[j] 是舊值還新值」。確認邏輯對了、再談壓維(follow-up)。s3[i+j−1] 那個索引是全題關鍵 —— 用掉 i+j 個字就填到 s3 第 i+j 位。

複雜度分析// COMPLEXITY

T = O(mn)　　S = O(n)(A) / O(mn)(B)

每格 O(1),m,n ≤ 100 → 至多 10⁴。A 只留一列 → O(s2.length),正是題目 follow-up 要的。

邊界與陷阱// PITFALLS

陷阱後果正解
忘記長度檢查s3 太長/太短卻誤判 true先 m+n != s3.size() → false
用 s3[i] 或 s3[j]索引錯 → 全錯是 s3[i+j−1]
只比字元、不看前一格字對上但前面根本接不起來兩條件 &&:dp[前一格] && 字元相等
1D 版忘記更新 dp[0]第一行(只用 s1)沒推進每列開頭 dp[0] = dp[0] && s1[i−1]==s3[i−1]

註:1D 版不需要像 1143 LCS 那樣留 prev —— 因為交錯只依賴上、左,沒有斜對角。這是兩題壓維難度的差別。
