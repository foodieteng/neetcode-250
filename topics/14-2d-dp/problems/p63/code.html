寫法 A · padding 哨兵(你的版本)// 1-INDEXED

p63.cpp C++17 
class Solution {
public:
    int uniquePathsWithObstacles(vector<vector<int>>& obstacleGrid) {
        int m = obstacleGrid.size(), n = obstacleGrid[0].size();
        vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0)); // 多一圈 0 哨兵

        if (obstacleGrid[0][0] || obstacleGrid[m - 1][n - 1]) return 0;

        dp[1][1] = 1;                        // 種子:對應真正起點 (0,0)

        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (obstacleGrid[i - 1][j - 1] == 1) continue; // 障礙 → 保持 0

                dp[i][j] = dp[i][j] + dp[i - 1][j] + dp[i][j - 1]; // 上 + 左
            }
        }

        return dp[m][n];
    }
};

dp 開 (m+1)×(n+1),第 0 列/行是全 0 哨兵 —— 這樣第一列/行的格子也能套同一條「上 + 左」而不必邊界特判(它們的上/左自動是 0)。dp[1][1]=1 是種子;障礙格 continue 保持 0。dp[i][j] += ... 的 dp[i][j] 起始是 0,寫成 += 或 = 都行。答案 dp[m][n]。

寫法 B · 一維滾動(O(n) 空間)// ROLLING

p63_1d.cpp C++17 
class Solution {
public:
    int uniquePathsWithObstacles(vector<vector<int>>& g) {
        int m = g.size(), n = g[0].size();
        vector<long long> dp(n, 0);
        dp[0] = 1;

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (g[i][j] == 1) dp[j] = 0;         // 障礙 → 0
                else if (j > 0) dp[j] += dp[j - 1]; // 上(舊 dp[j]) + 左(dp[j-1])
                // j==0 且非障礙:dp[0] 沿用(=上面的值)
            }
        }

        return (int)dp[n - 1];
    }
};

和 62 的一維版同源,只多兩件事:①障礙格把 dp[j]=0(截斷這條路);②第一欄 j==0 非障礙時 dp[0] 沿用上一列(它的「左」界外、只有「上」)。掃描仍由左而右。空間 O(n)。
