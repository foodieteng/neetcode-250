寫法 A · 二維(你的版本)// 2-D

p64.cpp C++17 
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<vector<int>> dp(m, vector<int>(n, INT_MAX));

        dp[0][0] = grid[0][0];
        for (int i = 1; i < n; i++)  // 第一列:前綴和
            dp[0][i] = dp[0][i - 1] + grid[0][i];
        for (int i = 1; i < m; i++)  // 第一行:前綴和
            dp[i][0] = dp[i - 1][0] + grid[i][0];

        for (int i = 1; i < m; i++)
            for (int j = 1; j < n; j++)
                dp[i][j] = min(dp[i - 1][j], dp[i][j - 1]) + grid[i][j];

        return dp[m - 1][n - 1];
    }
};

注意邊界不是 1、是前綴和:第一列只能一路往右、第一行只能一路往下,成本一路累加(這是與 62「全 1」最大的差)。內部格取 min(上, 左) 再加本格成本。答案右下角。

寫法 B · 一維滾動// O(n) SPACE

p64_1d.cpp C++17 
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size(), n = grid[0].size();
        vector<int> dp(n, 0);
        dp[0] = grid[0][0];
        for (int j = 1; j < n; j++) dp[j] = dp[j - 1] + grid[0][j]; // 第一列

        for (int i = 1; i < m; i++) {
            dp[0] += grid[i][0];                    // 第一行:只有「上」
            for (int j = 1; j < n; j++)
                dp[j] = min(dp[j], dp[j - 1]) + grid[i][j]; // dp[j]=上, dp[j-1]=左
        }
        return dp[n - 1];
    }
};

和 62 的一維版同源:dp[j](更新前)是上、dp[j-1] 是左,只是把 += 換成 min(dp[j], dp[j-1]) + 成本。第一行(j==0)只有「上」,故 dp[0] += grid[i][0]。掃描仍由左而右,空間 O(n)。
