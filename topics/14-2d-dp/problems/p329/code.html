寫法 A · 記憶化 DFS(最直觀)// TOP-DOWN

p329_dfs.cpp C++17 
class Solution {
public:
    int longestIncreasingPath(vector<vector<int>>& matrix) {
        const int m = matrix.size(), n = matrix[0].size();
        vector<vector<int>> memo(m, vector<int>(n, 0));   // dp 快取,0 = 還沒算

        int best = 0;
        for (int r = 0; r < m; ++r)
            for (int c = 0; c < n; ++c)
                best = max(best, dfs(r, c, matrix, memo));
        return best;
    }

private:
    const vector<pair<int, int>> DIR = {{1,0}, {-1,0}, {0,1}, {0,-1}};

    // 回傳從 (r,c) 出發的最長遞增路徑長
    int dfs(int r, int c, const vector<vector<int>>& A,
            vector<vector<int>>& memo) {
        if (memo[r][c]) return memo[r][c];       // 命中快取,直接回

        int m = A.size(), n = A[0].size();
        int best = 1;                          // 至少自己一格

        for (auto [dr, dc] : DIR) {
            int nr = r + dr, nc = c + dc;
            if (0 <= nr && nr < m && 0 <= nc && nc < n &&
                A[nr][nc] > A[r][c]) {          // 界內 && 嚴格更大
                best = max(best, 1 + dfs(nr, nc, A, memo));
            }
        }
        return memo[r][c] = best;            // 寫回快取
    }
};

不需要 visited 陣列 —— 因為只往「更大」走,天生不會回頭,環境自帶防環。 memo[r][c]==0 當「未算」的哨兵剛好安全,因為真實答案至少是 1,不會和 0 撞。

寫法 B · Kahn 拓撲分層(無遞迴)// BOTTOM-UP BFS

p329_kahn.cpp C++17 
class Solution {
public:
    int longestIncreasingPath(vector<vector<int>>& A) {
        const int m = A.size(), n = A[0].size();
        static const int DIR[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};

        // indeg[r][c] = 有幾個「更小的鄰居」指進來
        vector<vector<int>> indeg(m, vector<int>(n, 0));
        for (int r = 0; r < m; ++r)
            for (int c = 0; c < n; ++c)
                for (auto& d : DIR) {
                    int nr = r + d[0], nc = c + d[1];
                    if (0 <= nr && nr < m && 0 <= nc && nc < n &&
                        A[nr][nc] < A[r][c])
                        ++indeg[r][c];
                }

        // 所有源點(局部最小,indeg==0)入列
        queue<pair<int,int>> q;
        for (int r = 0; r < m; ++r)
            for (int c = 0; c < n; ++c)
                if (indeg[r][c] == 0) q.emplace(r, c);

        // 逐層剝;層數 = 最長路徑
        int layers = 0;
        while (!q.empty()) {
            ++layers;
            int sz = q.size();
            while (sz--) {
                auto [r, c] = q.front(); q.pop();
                for (auto& d : DIR) {
                    int nr = r + d[0], nc = c + d[1];
                    if (0 <= nr && nr < m && 0 <= nc && nc < n &&
                        A[nr][nc] > A[r][c])
                        if (--indeg[nr][nc] == 0) q.emplace(nr, nc);
                }
            }
        }
        return layers;
    }
};

把每格當節點、小 → 大連邊,這就是標準 Kahn 拓撲排序,只是不記順序、改數層數。 每剝一層 = 「所有更小鄰居都處理完」的格子放行。層數 = 最長鏈長,因為一個格子要等它最深的前驅剝完才輪到它。

A vs B · 怎麼選// TRADE-OFFS

A · 記憶化 DFSB · Kahn 分層
時間 / 空間O(mn) / O(mn)O(mn) / O(mn)
遞迴深度最壞 O(mn)(長蛇矩陣可能爆 stack)無遞迴,佇列迭代
程式碼量短,面試首選較長(算 indeg + BFS)
心智模型「往回收更大鄰居的答案」「剝洋蔥,層數即答案」
額外好處—若圖有環會剝不完 → 天然驗 DAG
面試講 A、加分講 B。A 記憶化更快寫、更好懂;能再補一句「這其實是 DAG 最長路徑,也能用拓撲排序自底向上、避免遞迴深度問題」—— 就展示了你看穿了問題的圖論本質,不只是套模板。

邊界與陷阱// PITFALLS

陷阱症狀正解
純 DFS 不 memo 指數時間,大矩陣 TLE 快取 dp[r][c],每格只算一次
用 >= 判遞增 相等也走 → 可能成環 → 無窮遞迴 / 答案爆大 必須嚴格 >
加了 visited 多餘,還可能擋掉合法重訪(不同起點共用尾巴) 嚴格遞增天生防環,不需要
memo 哨兵用 0 但答案可能是 0 本題不會 —— 路徑長至少 1,0 當「未算」安全 放心用 0;若題目答案可能 0 才要改哨兵
空矩陣 matrix[0] 越界 題目保證非空;面試口頭提一句即可
Kahn 邊方向搞反 indeg 數成「更大鄰居」→ 剝法錯 indeg = 更小鄰居數;relax 走向更大

細節走讀// WALKTHROUGH

// A · memo 表(從每格出發的最長路徑)—— 實跑 matrix memo(dp) 9 9 4 1 1 2 6 6 8 → 2 2 1 2 1 1 3 4 2 ^ 最大值 4 在值 1 那格 dp[值1] = 1 + dp[值2] = 1 + (1 + dp[值6]) = 1 + 1 + (1 + dp[值9]) = 1 + 1 + 1 + 1 = 4 ← 鏈 1→2→6→9 答案 = max(memo) = 4
// B · Kahn 逐層剝 —— 實跑 indeg(更小鄰居數) 層數(第幾層被剝) 1 2 0 4 3 1 1 1 3 → 3 2 3 1 0 0 2 1 1 層 1(源點 indeg=0):值 4、兩個 1 ← 局部最小 層 2:值 6(中)、值 2 層 3:值 6(左)、值 8、值 9(上) 層 4:值 9(左上角) 剝了 4 層 → 答案 = 4

兩張表算的是不同東西但答案一致:A 的 memo 是「往後還能走多長」,B 的層數是「到我為止最長走了多長」—— 一個從尾巴數、一個從頭數,最長鏈的總長相同。
